\begin{equation}
\begin{array}{lclcl}
T(2^{k+1}) & = & & 2T(2^{k+1}/2) + 2^{k+1} \\
 & = & & 2T(2^k) + 2^{k+1} \\
 & = & & 2(2^klg(2^k)) + 2^{k+1} \\
 & = & & 2^{k+1}lg(2^k) + 2^{k+1} \\
 & = & 2^{k+1}k + 2^{k+1} & &  2^{k+1}(lg(2^k) + 1) \\
 & = & 2^{k+1}(k + 1) & & 2^{k+1}(lg(2^k) + lg(2)) \\
 & = & 2^{k+1}(lg(2^{k + 1})) & & 2^{k+1}(lg(2^k \times 2)) \\
 & = & 2^{k+1}lg(2^{k+1}) & & 2^{k+1}lg(2^{k+1})
\end{array}
\end{equation}
